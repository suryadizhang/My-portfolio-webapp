### Backend (FastAPI + Postgres)
```python
# Booking endpoint with validation and rate limiting
@app.post("/api/bookings", response_model=BookingResponse)
@limiter.limit("5/minute")
async def create_booking(
    booking: BookingCreate,
    session: Session = Depends(get_session),
    current_user: User = Depends(get_current_user)
):
    # Validate availability
    if not await check_availability(session, booking.date, booking.duration):
        raise HTTPException(status_code=400, detail="Time slot not available")
    
    # Create booking with atomic transaction
    async with session.begin():
        db_booking = Booking(**booking.dict(), user_id=current_user.id)
        session.add(db_booking)
        await session.commit()
        
        # Send confirmation email
        await send_booking_confirmation(db_booking)
        
    return BookingResponse.from_orm(db_booking)

# Admin dashboard endpoints
@app.get("/api/admin/bookings")
@requires_role("admin")
async def get_all_bookings(
    page: int = 1,
    limit: int = 20,
    status: str = None,
    session: Session = Depends(get_session)
):
    query = select(Booking).order_by(Booking.created_at.desc())
    if status:
        query = query.where(Booking.status == status)
    
    bookings = await session.execute(
        query.offset((page - 1) * limit).limit(limit)
    )
    return {"bookings": bookings.scalars().all(), "page": page}
```

### Performance Optimizations
```typescript
// Optimistic UI updates for better UX
const [bookings, setBookings] = useState<Booking[]>([]);

const createBooking = async (bookingData: BookingCreate) => {
  // Optimistically add to UI
  const tempId = crypto.randomUUID();
  const optimisticBooking = { ...bookingData, id: tempId, status: 'pending' };
  setBookings(prev => [optimisticBooking, ...prev]);

  try {
    const result = await api.createBooking(bookingData);
    // Replace temp booking with real one
    setBookings(prev => 
      prev.map(b => b.id === tempId ? result : b)
    );
  } catch (error) {
    // Remove optimistic update on error
    setBookings(prev => prev.filter(b => b.id !== tempId));
    throw error;
  }
};
```

-- Bookings table with proper constraints
CREATE TABLE bookings (
    id UUID PRIMARY KEY DEFAULT gen_random_uuid(),
    user_id UUID REFERENCES users(id),
    event_date TIMESTAMP NOT NULL,
    duration INTEGER NOT NULL CHECK (duration > 0),
    guest_count INTEGER NOT NULL CHECK (guest_count > 0),
    services JSONB NOT NULL,
    total_amount DECIMAL(10,2) NOT NULL,
    status booking_status DEFAULT 'pending',
    created_at TIMESTAMP DEFAULT CURRENT_TIMESTAMP,
    updated_at TIMESTAMP DEFAULT CURRENT_TIMESTAMP
);
